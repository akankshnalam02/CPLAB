class Stack:
    def __init__(self):
        self.stack = []

    def push(self, item):
        self.stack.append(item)

    def pop(self):
        if not self.is_empty():
            return self.stack.pop()
        else:
            print("Stack is empty")

    def peek(self):
        if not self.is_empty():
            return self.stack[-1]
    
    def is_empty(self):
        return len(self.stack) == 0

# Example usage
s = Stack()
s.push(10)
s.push(20)
print(s.pop())
print(s.peek())
